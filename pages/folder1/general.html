                    public List getParsysComponentList(Page currentPage, String componentName) {
                        List componentList = new ArrayList();
                        if (currentPage != null && StringUtils.isNotBlank(componentName)) {
                            Resource pageResource = currentPage.getContentResource();
                            ResourceResolver resourceResolver = pageResource.getResourceResolver();
                            ResourceTreeIterator iterator = new ResourceTreeIterator(resourceResolver,
                                    pageResource.getPath(), new ResourceTreeIterator.Components());
                            while (iterator.hasNext()) {
                                Resource curComponent = iterator.next();
                                if (StringUtils.containsIgnoreCase(curComponent.getName(), componentName)) {
                                    componentList.add(curComponent.adaptTo(ValueMap.class));
                                }
                            }
                        }
                        return componentList;
                    }
